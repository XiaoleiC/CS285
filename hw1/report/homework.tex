\problem*{1}
\question{
Suppose $X \mid p \sim \operatorname{Binomial}(n, p)$ and $p \sim \operatorname{Beta}(\alpha, \beta)$. Suppose also that the loss function $L(p, \hat{p}(X))=(p-\hat{p}(X))^2$.
\begin{enumerate}[label=(\alph*)]
    \item Calculate the posterior risk $r(\hat{p} \mid X)$ for an arbitrary estimator $\hat{p}$.
    \item For a given $x$, what value of $\hat{p}(x)$ minimizes the posterior risk? Use this to construct a Bayes estimator.
    For this question do not rely on the general fact that the posterior mean minimizes squared error loss, but find the minimizer directly from minimizing the posterior risk from part a.
    \item Show that the posterior risk for the Bayes estimator you found in (b) is the posterior variance. Show that this going to be true generally for Bayes rules under squared error loss.
\end{enumerate}

}
\subproblem[1]

The posterior risk $r\left(\hat{p}\mid X\right)$ is
\begin{equation*}
    \begin{aligned}
        r\left(\hat{p}\mid X\right) & = \mathbb{E}\left(L\left(p,\hat{p}\right)\mid X\right) \\
        & = \int_{0}^{1} \left(p - \hat{p}\left(X\right)\right)^{2} f\left(p\mid X\right)\mathrm{d}p
    \end{aligned}
\end{equation*}
where $f\left(p\mid X\right)$ is the density of the $\operatorname{Beta}\left(\alpha+X,\beta + n - X\right)$.
